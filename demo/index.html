Basic math:

let a = new Vector(1, 1)["+"](1)["+"](0, 2); // currently: (2, 3)
const b = new Vector(1, 1);
a["+="](1, 1); // (3, 4)
a["+="]({x: 1, y: 1}); // (4, 5)
a["+="](b); // (5, 6)
a["-="](new Vector(1, 1)); // (4, 5)
a = a["+"](1, 1); // (5, 6)
console.log(a["+"](1, 1)); // output: (6, 7)
